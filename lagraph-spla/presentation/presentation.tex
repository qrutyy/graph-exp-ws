\begin{frame}
  \frametitle{BFS: первичные тезисы}
  \begin{itemize}
    \item GPU эффективнее на социальных графах (широкий фронтир, малый диаметр)
    \item На геометрических случайных графах (RGG) CPU выигрывает --- высокий диаметр,
          многократные запуски ядер с низкой загрузкой GPU
    \item LAGraph BFS на направленных графах требует кэша транспонированной матрицы
          (\texttt{LAGraph\_Cached\_AT}) для включения режима push+pull
    \item Первый запуск SPLA на GPU: 10--30× медленнее из-за JIT-компиляции ядер OpenCL
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{BFS: поддержка направленных/ненаправленных графов}
  \begin{block}{\texttt{LAGr\_BreadthFirstSearch}}
    Поддерживает оба режима через \texttt{LAGraph\_Kind}:
    \begin{itemize}
      \item Без \texttt{LAGraph\_Cached\_AT}: только push-фаза (медленнее)
      \item С \texttt{LAGraph\_Cached\_AT}: direction-optimising BFS (push + pull)
    \end{itemize}
  \end{block}
  \begin{block}{SPLA BFS}
    \texttt{--undirected=true/false}: симметризует матрицу перед построением графа
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{BFS: почему RGG невыгоден для GPU}
  \begin{itemize}
    \item RGG --- геометрический случайный граф: $\sim$4М вершин, $\sim$14 рёбер на вершину
    \item BFS имеет 1048 уровней
    \item Максимальный фронтир RGG: 7\,634 узла против 1.84М у LiveJournal
	\item Накладные расходы OpenCL
    \item SpLA directed: 706 мс vs LAGraph CPU: 534 мс --- CPU выигрывает на 32\%
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{BFS: выводы}
  \begin{itemize}
    \item \textbf{SpLA GPU быстрее} на социальных/коммуникационных графах (LJ: 4.8×, wiki-Talk: 1.9×)
    \item \textbf{CPU быстрее} на RGG --- высокий диаметр, малые фронтиры, низкая GPU-загрузка
    \item \textbf{Ненаправленный BFS быстрее} на социальных графах за счёт двунаправленных рёбер в pull-фазе
	\item BFS memory-bound: 10 потоков CPU (242 мс) медленнее 1 потока (202 мс) (OpenMP overhead)
    \item patents directed: топологический эффект --- 0.01\% вершин достижимо вперёд из max-degree-node
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{SSSP: первичные тезисы}
  \begin{itemize}
    \item LAGraph SSSP работает на обоих типах графов
    \item SPLA SSSP требует ненаправленный граф (push/pull нужна AT)
    \item GPU эффективен на плотных социальных графах, менее выгоден на разреженных
    \item Delta=1.0, unit-weights --- тестируем накладные расходы реализации
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{SSSP: поддержка направленных/ненаправленных}
  \begin{block}{LAGraph \texttt{LAGr\_SingleSourceShortestPath}}
    Поддерживает оба режима: directed, undirected
  \end{block}
  \begin{alertblock}{SPLA SSSP: только ненаправленные графы}
    SPLA SSSP использует push/pull-стратегию аналогично BFS.
    Для pull-фазы необходима транспонированная матрица AT.
    При \texttt{--undirected=false} результат может быть некорректным.
  \end{alertblock}
\end{frame}

\begin{frame}
  \frametitle{SSSP: wiki-Talk --- почему SPLA недоступен}
  \begin{alertblock}{SPLA SSSP зависает на wiki-Talk}
    Граф загружается успешно, вершина-источник выбирается, но вычисление
    не завершается даже за 5 минут.
  \end{alertblock}
  \vspace{0.4em}
  \textbf{Причина: экстремальное распределение степеней}
  \begin{itemize}
    \item 74\% вершин имеют степень 1 (после симметризации)
    \item Один хаб: степень 100\,029
    \item Средняя степень: 3.9 --- граф очень разреженный
  \end{itemize}
  \vspace{0.4em}
  \textbf{Как это ломает push/pull-переключение:}
  \begin{itemize}
    \item После первого раунда фронтир --- $\sim$100К соседей хаба
    \item Каждый из них имеет степень $\sim$1--2 $\Rightarrow$
  \end{itemize}
  \vspace{0.3em}
  \textit{LAGraph (CPU) справляется: 180 мс (directed), 126 мс (undirected)}
\end{frame}

\begin{frame}
  \frametitle{TC: первичные тезисы}
  \begin{itemize}
    \item GPU должен показывать многократное преимущество --- masked SpMM идеально ложится на архитектуру
    \item На разреженных графах (road) GPU менее эффективен из-за накладных расходов
    \item Предварительная сортировка вершин по степени важна для семейства Sandia на CPU ---
          но \textbf{только при правильных условиях}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{TC: алгоритмы Sandia}
  Семейство Sandia использует разбиение на нижнетреугольную (L) и верхнетреугольную (U) части:
  \begin{itemize}
    \item \textbf{Sandia\_LL}: $\text{ntri} = \sum L \cdot (L \times L)$ --- нижняя $\times$ нижняя
    \item \textbf{Sandia\_LUT}: $\text{ntri} = \sum L \cdot (L \times U^T)$ --- нижняя $\times$ верхняя$^T$
    \item \textbf{Burkhardt}: $\text{ntri} = \frac{1}{6}\sum (A^2 \cdot A)$ --- полная матрица
    \item \textbf{Cohen}: через пересечение списков смежности
  \end{itemize}

  Ключевая операция: для каждого ребра $(i,j)$ в $L$ считаем пересечение $L[i,:] \cap L[j,:]$.
\end{frame}

\begin{frame}
  \frametitle{TC: почему сортировка должна помогать}
  \textbf{Аргумент стоимости пересечения:}
  \begin{itemize}
    \item При сортировке по возрастанию степени: строка $i$ в $L$ имеет $\leq d(i)$ ненулей
    \item Вершины с малой степенью --- в начале матрицы, пересечения их списков дёшевы
    \item Стоимость пересечения $\approx O(d(i) + d(j))$ --- минимизируется когда оба малы
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{TC: эвристика AutoSort}
  AutoSort LAGraph включает сортировку если выполнены \textbf{все} условия:
  \begin{itemize}
    \item $n > 2000$ --- граф достаточно большой для амортизации стоимости
    \item $\text{density} > 10$ --- средняя степень достаточна для выгоды
    \item $\text{mean} > 3 \times \text{median}$ --- степенной закон (наличие хабов)
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{TC: код сортировки (LAGraph source)}
  \begin{Verbatim}[fontsize=\scriptsize]
if (presort != LAGr_TriangleCount_NoSort) {
    // Step 1: compute degree permutation P
    LAGr_SortByDegree(&P, G, true,
        presort == LAGr_TriangleCount_Ascending, msg);

    // Step 2: permute A -> T = A(P,P)  -- O(nnz), cache-heavy
    GrB_Matrix_new(&T, GrB_BOOL, n, n);
    GrB_extract(T, NULL, NULL, A,
        (GrB_Index*)P, n, (GrB_Index*)P, n, NULL);
    A = T;
}

// Step 3: L = tril(A), then count = reduce(L .* (L*L))
  \end{Verbatim}
  \begin{itemize}
    \item \texttt{GrB\_extract} с индексными массивами = случайная перестановка nnz записей
    \item Именно эта операция съедает 2.36с при любой сортировке
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{BFS: direction-optimizing алгоритм (SPLA)}
  \begin{itemize}
    \item Push: обходит граф по направлению рёбер (фронтир $\rightarrow$ соседи)
    \item Pull: обходит граф против направления (непосещённые $\rightarrow$ фронтир через AT)
    \item Переключение по плотности текущего фронтира
  \end{itemize}
\begin{Verbatim}[fontsize=\scriptsize]
if (push || (push_pull && is_push_better)) {
    exec_vxm_masked(frontier_new, v, frontier_prev, A, ...);
} else {
    exec_mxv_masked(frontier_new, v, A, frontier_prev, ...);
}
\end{Verbatim}
  \begin{itemize}
    \item Для Pull на направленных графах необходима AT (\texttt{LAGraph\_Cached\_AT})
    \item Без AT: только push-фаза, алгоритм медленнее на плотных фронтирах
  \end{itemize}
\end{frame}
